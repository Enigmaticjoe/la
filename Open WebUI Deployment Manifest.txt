4. PHASE II: INTELLIGENCE ACQUISITION
Objective: Download the specific AI models optimized for your hardware. We use a temporary Docker container to avoid dependency hell on the host OS.
Instructions:
1. Create a new User Script named VOID_FETCH.
2. Paste the code below.
3. Run in Background (This will take time depending on your internet speed).
4. Monitor: Check logs by clicking the "Log" button on the script. Wait for "ASSETS SECURED".


Bash




#!/bin/bash
# SCRIPT: VOID_FETCH
MODEL_DIR="/mnt/user/appdata/brain_ai/models"

# --- THE PAYLOAD ---
# LLM: Dolphin 2.9.4 (8B) - Uncensored, fast, fits in 12GB VRAM.
LLM_REPO="solidrust/dolphin-2.9.4-llama3.1-8b-AWQ"
# EMBED: Qwen 2.5 - High performance semantic understanding.
EMBED_REPO="Qwen/Qwen2.5-Embedding"
# RERANK: BAAI Large - Critical for RAG accuracy.
RERANK_REPO="BAAI/bge-reranker-large"

echo ">>> [VOID] DEPLOYING DOWNLOADER DRONE..."

docker run --rm \
 -v "$MODEL_DIR":/data \
 -e HF_HUB_ENABLE_HF_TRANSFER=1 \
 python:3.11-slim \
 bash -c "pip install -q huggingface_hub[cli] hf_transfer && \
 echo '>>> [1/3] DOWNLOADING LLM...' && \
 huggingface-cli download $LLM_REPO --local-dir /data/llm --local-dir-use-symlinks False --include '*.json' '*.safetensors' '*.model' && \
 echo '>>> [2/3] DOWNLOADING EMBEDDER...' && \
 huggingface-cli download $EMBED_REPO --local-dir /data/embed --local-dir-use-symlinks False && \
 echo '>>> [3/3] DOWNLOADING RERANKER...' && \
 huggingface-cli download $RERANK_REPO --local-dir /data/rerank --local-dir-use-symlinks False"

echo ">>> [VOID] ASSETS SECURED. PERMISSION FIX APPLIED."
chown -R 99:100 "$MODEL_DIR"

5. PHASE III: THE GOLDEN STACK (DEPLOYMENT)
Objective: Spin up the infrastructure using Portainer.
Instructions:
1. Open Portainer > Stacks > Add stack.
2. Name: brain_ai_stack
3. Paste the YAML below.
4. Click Deploy the stack.


YAML




version: '3.8'

services:
 # [THE ENGINE] vLLM: Serves the LLM API
 vllm:
   image: vllm/vllm-openai:v0.15.1
   container_name: brain_vllm
   runtime: nvidia
   deploy:
     resources:
       reservations:
         devices:
           - driver: nvidia
             count: 1
             capabilities: [gpu]
   environment:
     - NVIDIA_VISIBLE_DEVICES=all
     - NCCL_P2P_DISABLE=1
   volumes:
     - /mnt/user/appdata/brain_ai/models/llm:/model:ro
   command: >
     --model /model
     --quantization awq_marlin
     --gpu-memory-utilization 0.60
     --max-model-len 8192
     --enforce-eager
     --served-model-name dolphin-8b
     --port 8000
    ports:
     - "8000:8000"
   ipc: host
   networks:
     - ai_grid

 # [THE BRAIN] Open WebUI: The Chat Interface
 open-webui:
   image: ghcr.io/open-webui/open-webui:main
   container_name: brain_webui
   restart: unless-stopped
   ports:
     - "3000:8080"
   environment:
     - OPENAI_API_BASE_URL=http://vllm:8000/v1
     - OPENAI_API_KEY=sk-void-key
     - WEBUI_SECRET_KEY=JULES_PROTOCOL_SECURE_KEY
     # RAG Configuration (Retrieval Augmented Generation)
     - ENABLE_RAG_WEB_SEARCH=True
     - RAG_EMBEDDING_ENGINE=openai
     - RAG_OPENAI_API_BASE_URL=http://tei-embed:80/v1
     - RAG_OPENAI_API_KEY=sk-void
     - RAG_RERANKING_MODEL=BAAI/bge-reranker-large
     - RAG_EXTERNAL_RERANKER_URL=http://tei-rerank:80/rerank
     - RAG_EXTERNAL_RERANKER_API_KEY=sk-void
     - VECTOR_DB_URL=http://qdrant:6333
   volumes:
     - /mnt/user/appdata/brain_ai/open-webui:/app/backend/data

     - ai_grid

 # [SEMANTIC MEMORY] Text Embeddings Interface
