# Brain Node - Python Dependencies for AMD ROCm

# on AMD Radeon RX 7900 XT using ROCm and vLLM.

#   pip install -r brain-requirements.txt

#   python3 -m venv ~/venv-brain
#   source ~/venv-brain/bin/activate
#   pip install -r brain-requirements.txt

# ============================================================================
# Installation Notes for AMD ROCm
# ============================================================================

# 1. INSTALL ROCM FIRST
# ----------------------
# Follow BRAIN-AMD-SETUP.md for ROCm installation
# ROCm 6.2+ required for RX 7900 XT support on Ubuntu 24.04
# ROCm 6.0+ works for Ubuntu 22.04
#
# Ubuntu 24.04 (Noble):
#   sudo mkdir --parents --mode=0755 /etc/apt/keyrings
#   wget https://repo.radeon.com/rocm/rocm.gpg.key -O - | gpg --dearmor | sudo tee /etc/apt/keyrings/rocm.gpg > /dev/null
#   echo "deb [arch=amd64 signed-by=/etc/apt/keyrings/rocm.gpg] https://repo.radeon.com/rocm/apt/latest noble main" | sudo tee /etc/apt/sources.list.d/rocm.list
#   sudo apt update
#   sudo apt install rocm-hip-sdk rocm-libs
#
# Ubuntu 22.04 (Jammy):
#   sudo mkdir --parents --mode=0755 /etc/apt/keyrings
#   wget https://repo.radeon.com/rocm/rocm.gpg.key -O - | gpg --dearmor | sudo tee /etc/apt/keyrings/rocm.gpg > /dev/null
#   echo "deb [arch=amd64 signed-by=/etc/apt/keyrings/rocm.gpg] https://repo.radeon.com/rocm/apt/latest jammy main" | sudo tee /etc/apt/sources.list.d/rocm.list
#   sudo apt update
#   sudo apt install rocm-hip-sdk rocm-libs

# 2. INSTALL PYTORCH FOR ROCM
# ----------------------------
# DO NOT use: pip install torch
# Use ROCm-specific version:
#
# For ROCm 6.2+ (Ubuntu 24.04):
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/rocm6.2
#
# For ROCm 6.1 (Ubuntu 22.04):
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/rocm6.1
#
# For ROCm 6.0 (older):
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/rocm6.0
#
# Verify installation:
#   python3 -c "import torch; print(torch.cuda.is_available()); print(torch.version.hip)"

# 3. VLLM FOR AMD ROCM
# --------------------
# vLLM runs in Docker container (brain-stack.yml)
# Uses image: vllm/vllm-openai:v0.15.1
# ROCm support is built into the official vLLM Docker images
#
# For local vLLM installation (advanced, not needed for Docker setup):
#   pip install vllm
#
# Note: Local vLLM may require building from source for optimal AMD support

# 4. AUTOAWQ FOR AMD (Optional - for local testing)
# --------------------------------------------------
# AutoAWQ provides AWQ quantization support
# AMD ROCm support is experimental
#
# Installation:
#   pip install autoawq
#
# Or build from source for latest AMD optimizations:
#   git clone https://github.com/casper-hansen/AutoAWQ.git
#   cd AutoAWQ
#   pip install -e .

# 5. ALTERNATIVE: VLLM (If running outside Docker)
# -------------------------------------------------
# vLLM natively supports AWQ models and is recommended over AutoAWQ
# for inference on AMD GPUs. It's already included in the Docker stack.

# ============================================================================
# Recommended Installation Steps
# ============================================================================
#
# Step 1: Create virtual environment
#   python3 -m venv ~/venv-brain
#   source ~/venv-brain/bin/activate
#
# Step 2: Upgrade pip
#   pip install --upgrade pip setuptools wheel
#
# Step 3: Install PyTorch for ROCm
# For Ubuntu 24.04 with ROCm 6.2+:
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/rocm6.2
#
# For Ubuntu 22.04 with ROCm 6.1:
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/rocm6.1
#
# For older ROCm 6.0:
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/rocm6.0
#
# Step 4: Install this requirements file
#   pip install -r brain-requirements.txt
#
# Step 5: Verify installation
#   python3 -c "import torch; print(f'PyTorch: {torch.__version__}'); print(f'ROCm available: {torch.cuda.is_available()}')"
#   huggingface-cli --version
#
# Step 6: Download models (optional, can be done in Docker)
#   bash brain-model-downloader.sh
